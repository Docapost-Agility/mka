let getContextMenu = (selection) => {
    let menu = [];

    menu.push({
        title: "action 1",
        action: () => {
            console.log("action 1");
        }
    });
    menu.push({
        title: "action 3",
        action: () => {
            console.log("action 3");
        }
    });
    if (selection.length > 1) {
        menu.push({
            title: "action multiple",
            action: () => {
                console.log("action multiple");
            }
        });
    }
    return menu;
}
